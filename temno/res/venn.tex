	\begin{tikzpicture}
		\def\firstcircle{(0,0) circle (1.5 cm)}
		\def\secondcircle{(0: 2 cm) circle (1.5 cm)}
		\def\thirdcircle{(60:2 cm) circle (1.5 cm)}
		
		% Draw the circles
		\draw[color=myblue] \firstcircle;

		\end{scope}
		
		
		\begin{scope}
			\clip \firstcircle;
			\fill[pattern=horizontal lines, pattern color=myblue] \secondcircle;
		\end{scope}

		\node[left] at (0,0) {$A$};
